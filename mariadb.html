MariaDB en Docker

>>>> En el anfitrión
docker run -d -p 7706:3306 --name mymariadb --network bridge -e MYSQL_ROOT_PASSWORD=toor mariadb

>>>> En el contenedor
# mysql -uroot -ptoor
MariaDB [(none)]> use mysql
MariaDB [mysql]> GRANT ALL PRIVILEGES ON *.* TO 'remoto'@'%' IDENTIFIED BY 'remoto';
MariaDB [mysql]> FLUSH PRIVILEGES;
MariaDB [mysql]> \q

>>>> En el anfitrión Linux
$ ifconfig o ip address ---> ipDocker

$ mysql -u remoto -p -h <ipDocker> -P 7706

>>>> En el anfitrión Windows
PC C:\> ipconfig ---> ipWSL

PC C:\> \xampp\mysql\bin\mysql -u remoto -p -h <ipWSL> -P 7706
